(require 'eww)

(defun test-servlet-call (servlet-name &optional parse-headers)
  (let* ((temp-buffer
          (url-retrieve-synchronously
           (format "http://firstthreeodds.org/%s" servlet-name) t t))
         (reply ""))
    (with-current-buffer temp-buffer
      (setq reply (buffer-string)) ;; not necessary
      (when parse-headers
        (unwind-protect
            (let* ((headers (setq hs (eww-parse-headers)))
                   (content-length (setq cl (string-to-number (cdr (assoc "content-length" headers)))))
                   (content (buffer-substring (point) (point-max))))
              (setq reply content)))))
    (kill-buffer temp-buffer)
    reply))
